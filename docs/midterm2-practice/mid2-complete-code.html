import numpy as np

a, b    = 0.0, 3.0
exact   = 0.88621
rng     = np.random.default_rng(seed=42)

def f(x):
    return np.exp(-x**2)

for N in [1_000, 10_000, 100_000]:
    x_samples = rng.uniform(a, b, N)
    I_mc      = (b - a) * np.mean(f(x_samples))
    error     = abs(I_mc - exact)
    print(f"N = {N:>8,d}  |  I_mc = {I_mc:.5f}  |  error = {error:.5f}")
